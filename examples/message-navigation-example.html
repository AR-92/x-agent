Usage Example

import { MessageNavigation } from './components/MessageNavigation.js';

// Create navigation instance
const navigation = new MessageNavigation({
  container: document.getElementById('navContainer'),
  currentIndex: 0,
  totalMessages: 5,
  isFirstMessage: false,
  isLastMessage: true,
  onNavigate: (index) => {
    // Handle navigation
    const message = document.querySelector(\`[data-message-index="\${index}"]\`);
    if (message) {
      message.scrollIntoView({ behavior: 'smooth', block: 'center' });
    }
  },
});

// Update navigation state
navigation.update({ currentIndex: 2, totalMessages: 6 });
